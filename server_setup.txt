#### NGINX Config /etc/nginx/conf.d/default.conf ####

upstream pdfunicorn {
  server 127.0.0.1:8080;
}

server {
    listen 80;
    server_name www.pdfunicorn.com pdfunicorn.com;
    return 301 https://pdfunicorn.com$request_uri;
}

server {
    listen 443;
    server_name www.pdfunicorn.com;
    return 301 https://pdfunicorn.com$request_uri;
}

server {
  listen 443 default_server ssl;
  server_name localhost;
  ssl_certificate      /etc/ssl/localcerts/pdfunicorn.com.crt;
  ssl_certificate_key  /etc/ssl/localcerts/pdfunicorn.com.key;
  ssl_verify_depth 3;
  # http://blog.visionsource.org/2011/06/08/setting-up-thawte-ssl-123-on-nginx/
  # https://ssltools.thawte.com/checker/views/certCheck.jsp

  ssl_session_cache shared:SSL:1m;
  ssl_session_timeout  5m;

  ssl_ciphers  HIGH:!aNULL:!MD5;
  ssl_prefer_server_ciphers   on;

  location / {
    proxy_pass http://pdfunicorn;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
    proxy_set_header Host $host;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-HTTPS 0;
  }
}
